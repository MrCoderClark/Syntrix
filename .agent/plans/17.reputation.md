# Reputation & Badges Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add a reputation system that awards points for votes received and answers accepted, a badge system with milestone achievements, and display both on user profiles.

**Architecture:** Reputation is event-sourced — `reputation_events` is the append-only log, `users.reputation` is a trigger-maintained denormalized total. Badges are checked lazily on relevant actions (vote, accept, create answer/question) and inserted into `user_badges` when criteria are met. Both reputation and badges display on the profile page. The vote and answer-accept endpoints are extended to emit rep events and check badge criteria after each action.

**Tech Stack:** PostgreSQL triggers (score maintenance), SQLAlchemy Core (queries), FastAPI (endpoints), React/TypeScript (profile UI)

## Reputation Point Values

| Event Type | Delta | Reverse | Trigger |
|---|---|---|---|
| `answer_upvoted` | +10 | -10 on unvote | someone upvotes your answer |
| `answer_downvoted` | -2 | +2 on unvote | someone downvotes your answer |
| `answer_accepted` | +15 | -15 on unaccept | your answer is accepted |
| `accept_answer` | +2 | -2 on unaccept | you accept an answer (asker reward) |
| `question_upvoted` | +5 | -5 on unvote | someone upvotes your question |
| `question_downvoted` | -2 | +2 on unvote | someone downvotes your question |
| `downvote_answer_cost` | -1 | +1 on unvote | you downvote an answer (cost) |

# backend/app/reputation/badges.py
from __future__ import annotations

import uuid

from sqlalchemy import func, select
from sqlalchemy.ext.asyncio import AsyncSession

from app.models import Answer, Post
from app.models.reputation import Badge, UserBadge


async def check_badges(
    session: AsyncSession,
    user_id: uuid.UUID,
) -> list[Badge]:
    """Check all badge criteria for a user and award any newly earned badges.

    Returns list of newly awarded badges (empty if none).
    """
    existing_q = select(UserBadge.badge_id).where(
        UserBadge.user_id == user_id
    )
    existing_result = await session.execute(existing_q)
    owned_ids = {row[0] for row in existing_result}

    all_badges_result = await session.execute(select(Badge))
    all_badges = list(all_badges_result.scalars())

    newly_awarded: list[Badge] = []
    for badge in all_badges:
        if badge.id in owned_ids:
            continue
        if await _meets_criteria(session, user_id, badge.criteria):
            session.add(
                UserBadge(user_id=user_id, badge_id=badge.id)
            )
            newly_awarded.append(badge)

    return newly_awarded


async def _meets_criteria(
    session: AsyncSession,
    user_id: uuid.UUID,
    criteria: dict,
) -> bool:
    ctype = criteria["type"]
    threshold = criteria["threshold"]

    if ctype == "question_count":
        q = (
            select(func.count())
            .select_from(Post)
            .where(
                Post.author_id == user_id,
                Post.post_type == "question",
                Post.deleted_at.is_(None),
            )
        )
        result = await session.execute(q)
        return result.scalar_one() >= threshold

    if ctype == "answer_count":
        q = (
            select(func.count())
            .select_from(Answer)
            .where(
                Answer.author_id == user_id,
                Answer.deleted_at.is_(None),
            )
        )
        result = await session.execute(q)
        return result.scalar_one() >= threshold

    if ctype == "accepted_answer_count":
        q = (
            select(func.count())
            .select_from(Answer)
            .where(
                Answer.author_id == user_id,
                Answer.is_accepted.is_(True),
                Answer.deleted_at.is_(None),
            )
        )
        result = await session.execute(q)
        return result.scalar_one() >= threshold

    if ctype == "questions_with_accepted":
        q = (
            select(func.count())
            .select_from(Post)
            .where(
                Post.author_id == user_id,
                Post.post_type == "question",
                Post.has_accepted_answer.is_(True),
                Post.deleted_at.is_(None),
            )
        )
        result = await session.execute(q)
        return result.scalar_one() >= threshold

    if ctype == "answer_score_gte":
        q = (
            select(func.count())
            .select_from(Answer)
            .where(
                Answer.author_id == user_id,
                Answer.score >= threshold,
                Answer.deleted_at.is_(None),
            )
        )
        result = await session.execute(q)
        return result.scalar_one() >= 1

    if ctype == "questions_with_score_gte":
        min_score = criteria.get("min_score", 1)
        q = (
            select(func.count())
            .select_from(Post)
            .where(
                Post.author_id == user_id,
                Post.post_type == "question",
                Post.score >= min_score,
                Post.deleted_at.is_(None),
            )
        )
        result = await session.execute(q)
        return result.scalar_one() >= threshold

    return False
```

# backend/app/reputation/engine.py
from __future__ import annotations

import uuid

from sqlalchemy.ext.asyncio import AsyncSession

from app.models.reputation import ReputationEvent
from app.reputation.badges import check_badges

REP_VALUES = {
    "answer_upvoted": 10,
    "answer_downvoted": -2,
    "answer_accepted": 15,
    "accept_answer": 2,
    "question_upvoted": 5,
    "question_downvoted": -2,
    "downvote_answer_cost": -1,
}


async def award_rep(
    session: AsyncSession,
    user_id: uuid.UUID,
    event_type: str,
    source_id: uuid.UUID | None = None,
) -> None:
    """Insert a reputation event. The DB trigger updates users.reputation."""
    delta = REP_VALUES.get(event_type)
    if delta is None or delta == 0:
        return

    event = ReputationEvent(
        user_id=user_id,
        event_type=event_type,
        delta=delta,
        source_id=source_id,
    )
    session.add(event)
    await session.flush()
    await check_badges(session, user_id)
```

## Task 4: Wire rep events into vote endpoints

**Files:**

- Modify: `backend/app/answers/router.py`

The reputation engine must be called after each vote action. The key logic:

- **Post upvote/downvote (question only)**: award rep to the post *author* (not the voter)
- **Answer upvote**: award rep to the answer *author* (+10), and if downvote, also charge the *voter* (-1 cost)
- **Answer downvote**: penalize answer author (-2), charge voter (-1 cost)
- **Unvote (value=0)**: reverse the previous event — this is done by inserting a new event with the negated delta type

For simplicity, we track reversal by inserting a counter-event. The trigger sums deltas.

async def award_rep(
    session: AsyncSession,
    user_id: uuid.UUID,
    event_type: str,
    source_id: uuid.UUID | None = None,
    *,
    reverse: bool = False,
) -> None:
    """Insert a reputation event. The DB trigger updates users.reputation.

    If reverse=True, the delta is negated (used when unvoting/unaccepting).
    """
    delta = REP_VALUES.get(event_type)
    if delta is None or delta == 0:
        return

    if reverse:
        delta = -delta

    event = ReputationEvent(
        user_id=user_id,
        event_type=event_type,
        delta=delta,
        source_id=source_id,
    )
    session.add(event)
    await session.flush()
    if not reverse:
        await check_badges(session, user_id)
```

from __future__ import annotations

from fastapi import APIRouter, Depends, HTTPException, Query
from sqlalchemy import select
from sqlalchemy.ext.asyncio import AsyncSession

from app.db.session import get_session
from app.models import User
from app.models.reputation import Badge, ReputationEvent, UserBadge

from .schemas import (
    BadgeResponse,
    RepEventResponse,
    UserBadgeResponse,
    UserRepResponse,
)

router = APIRouter(prefix="/api/users", tags=["reputation"])


@router.get("/{handle}/reputation", response_model=UserRepResponse)
async def get_user_reputation(
    handle: str,
    limit: int = Query(default=50, le=100),
    session: AsyncSession = Depends(get_session),
):
    result = await session.execute(
        select(User).where(User.handle == handle)
    )
    user = result.scalar_one_or_none()
    if not user:
        raise HTTPException(status_code=404, detail="User not found")

    events_q = (
        select(ReputationEvent)
        .where(ReputationEvent.user_id == user.id)
        .order_by(ReputationEvent.created_at.desc())
        .limit(limit)
    )
    events_result = await session.execute(events_q)
    events = events_result.scalars().all()

    badges_q = (
        select(UserBadge, Badge)
        .join(Badge, UserBadge.badge_id == Badge.id)
        .where(UserBadge.user_id == user.id)
        .order_by(UserBadge.awarded_at.desc())
    )
    badges_result = await session.execute(badges_q)
    badge_rows = badges_result.all()

    return UserRepResponse(
        reputation=user.reputation,
        events=[
            RepEventResponse(
                id=str(e.id),
                event_type=e.event_type,
                delta=e.delta,
                source_id=str(e.source_id) if e.source_id else None,
                created_at=e.created_at,
            )
            for e in events
        ],
        badges=[
            UserBadgeResponse(
                badge=BadgeResponse(
                    slug=row.Badge.slug,
                    name=row.Badge.name,
                    description=row.Badge.description,
                    icon=row.Badge.icon,
                    tier=row.Badge.tier,
                ),
                awarded_at=row.UserBadge.awarded_at,
            )
            for row in badge_rows
        ],
    )
```

async def test_answer_vote_awards_rep_via_api(
    db_conn: AsyncConnection,

        {
            "id": comm_id,

            "s": comm_slug,
            "d": "test",
            "c": "#000",
            "cb": admin["id"],
        },
    )
    # Add memberships
    for u in [author, answerer, voter]:
        await db_conn.execute(
            text(
                "INSERT INTO syntrix.community_memberships "
                "(community_id, user_id, role) "
                "VALUES (:cid, :uid, 'member')"
            ),
            {"cid": comm_id, "uid": u["id"]},
        )
    await db_conn.commit()

    transport = ASGITransport(app=app)
    async with AsyncClient(
        transport=transport, base_url="http://test"
    ) as client:
        # Author creates a question
        q_resp = await client.post(
            "/api/posts",
            json={
                "community_id": str(comm_id),
                "title": "Rep test question?",
                "body_json": {"type": "doc", "content": []},
                "post_type": "question",
            },
            cookies={"access_token": author["token"]},
        )
        assert q_resp.status_code == 201
        q_id = q_resp.json()["id"]

        # Answerer posts an answer
        a_resp = await client.post(
            f"/api/posts/{q_id}/answers",
            json={
                "body_json": {"type": "doc", "content": []},
            },
            cookies={"access_token": answerer["token"]},
        )
        assert a_resp.status_code == 201
        a_id = a_resp.json()["id"]

        # Voter upvotes the answer
        v_resp = await client.post(
            f"/api/answers/{a_id}/vote",
            json={"value": 1},
            cookies={"access_token": voter["token"]},
        )
        assert v_resp.status_code == 200

        # Check answerer's reputation went up by 10
        rep_resp = await client.get(
            f"/api/users/{answerer['handle']}/reputation"
        )
        assert rep_resp.status_code == 200
        assert rep_resp.json()["reputation"] == 10

        # Voter unvotes
        v_resp2 = await client.post(
            f"/api/answers/{a_id}/vote",
            json={"value": 0},
            cookies={"access_token": voter["token"]},
        )
        assert v_resp2.status_code == 200

        # Reputation back to 0
        rep_resp2 = await client.get(
            f"/api/users/{answerer['handle']}/reputation"
        )
        assert rep_resp2.json()["reputation"] == 0
